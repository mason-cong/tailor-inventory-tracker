☁️ Google Sheets Sync

Back up all your orders to a Google Sheet so data is never lost. You'll need to create a free Google Apps Script web app — it takes about 5 minutes.

  1. Open Google Sheets and create a new spreadsheet
  2. Go to Extensions → Apps Script
  3. Delete any existing code and paste in the script below
  4. Click Deploy → New deployment → Web app
  5. Set "Who has access" to Anyone, then click Deploy
  6. Copy the Web App URL and paste it below

Apps Script code to paste:

function doPost(e){var ss=SpreadsheetApp.getActiveSpreadsheet();var sh=ss.getSheetByName('Orders')||ss.insertSheet('Orders');var data=JSON.parse(e.postData.contents);if(sh.getLastRow()===0)sh.appendRow(['ID','Customer','Contact','Items','Price','Payment','Due Date','Status','Notes','Photos','Created']);data.orders.forEach(function(o){var existing=sh.getDataRange().getValues().find(function(r){return r[0]==o.id});if(!existing)sh.appendRow([o.id,o.name,o.contact||'',o.items.map(function(i){return i.qty+'x '+i.type}).join(', '),o.price,o.payment,o.due,o.status,o.notes||'',(o.photos||[]).length+' photo(s)',o.createdAt])});return ContentService.createTextOutput(JSON.stringify({ok:true,synced:data.orders.length})).setMimeType(ContentService.MimeType.JSON)}
